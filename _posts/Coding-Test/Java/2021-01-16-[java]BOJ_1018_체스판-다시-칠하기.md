public class Main {
    public static void main(String[] args) throws Exception {
        Solution solution = new Solution();
        solution.solve();
    }
}

class solution {
    public static Scanner sc = new Scanner(System.in);

    private int N, M, min;
    private String[][] board;

    private int colorCount(int n, int m) {
        int[] whiteBlackCount = {0, 0};
        String[] color = {"W", "B"};
        int colorIndex;
        for (int c=0; c < 2; c++) {
            for (int i=0; i < 8; i++) {
                for (int j=0; j < 8; j++) {
                    colorIndex = (c+i+j) % 2;
                    if (!board[n+i][m+j].equals(color[colorIndex])) {
                        whiteBlackCount[c] += 1;
                    }
                }
            }
        }

        return whiteBlackCount[0] < whiteBlackCount[1] ? whiteBlackCount[0] : whiteBlackCount[1];
    }
    
    public void solve() {
        N = Integer.parseInt(sc.next());
        M = Integer.parseInt(sc.next());
        board = new String[N][M];

        for (int n=0; n < N; n++) {
            board[n] = sc.next().split("");
        }

        min = 64;
        int count;
        for (int n=0; n+7 < N; n++) {
            for (int m=0; m+7 < M; m++) {
                count = colorCount(n, m);
                if (count < min) {
                    min = count;
                }
            }
        }

        System.out.println(min);
    }
}
