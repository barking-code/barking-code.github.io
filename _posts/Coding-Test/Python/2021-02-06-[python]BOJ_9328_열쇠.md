[BOJ 9328 열쇠](https://www.acmicpc.net/problem/9328)

```python
from sys import stdin
from collections import deque

dr = [-1, 0, 1, 0]
dc = [0, 1, 0, -1]

def solution():
    def open_door(key):
        for r, c in doors[key.upper()]:
            board[r][c] = "."
        return

    R, C = map(int, stdin.readline().split())
    board = [list(stdin.readline().strip()) for _ in range(R)]
    tmp = stdin.readline().strip()
    
    keys = [0] * 91
    if tmp != "0":
        for k in list(tmp):
            keys[ord(k.upper())] = 1
    
    # 키 줍기 + 문서 줍기
    doc_count = 0
    for r in range(R):
        for c in range(C):
            if r == 0 or r == R-1 or c == 0 or c ==C-1:
                if 97 <= ord(board[r][c]) <= 122:
                    keys[ord(board[r][c].upper())] = 1
                elif board[r][c] == "$":
                    doc_count += 1
                else:
                    continue

                board[r][c] = "."

    # 시작할 수 있는 위치 찾기 + 키를 가지고 있는 문 열어놓기
    cors = deque()
    doors = {chr(k): [] for k in range(65, 91)}
    for r in range(R):
        for c in range(C):
            if 65 <= ord(board[r][c]) <= 90 and keys[ord(board[r][c])]:
                board[r][c] = "."
            
            if r == 0 or r == R-1 or c == 0 or c ==C-1:
                if board[r][c] == ".":
                    pass
                elif 65 <= ord(board[r][c]) <= 90:
                    doors[board[r][c]].append((r, c))
                    continue
                else:
                    continue

                cors.append((r, c))
    
    visit = [[0 for _ in range(C)] for _ in range(R)]
    for r, c in cors:
        visit[r][c] = 1

    while cors:
        r, c = cors.popleft()
        for d in range(4):
            tr, tc = r+dr[d], c+dc[d]

            if tr < 0 or tr > R-1 or tc < 0 or tc > C-1: continue
            if visit[tr][tc]: continue

            tile = board[tr][tc]

            if tile == "*":
                continue
            elif tile == ".":
                pass
            elif tile == "$":
                board[tr][tc] = "."
                doc_count += 1
            elif 97 <= ord(tile) <= 122:
                board[tr][tc] = "."
                keys[ord(tile.upper())] = 1

                open_door(tile)

                for rr, cc in doors[tile.upper()]:
                    visit[rr][cc] = 1
                    cors.append((rr, cc))
                
            elif 65 <= ord(tile) <= 90:
                if keys[ord(tile.upper())]:
                    board[tr][tc] = "."
                else:
                    doors[tile].append((tr, tc))
                    continue

            cors.append((tr, tc))
            visit[tr][tc] = 1
    
    print(doc_count)


T = int(stdin.readline())
for t in range(T):
    solution()
```
